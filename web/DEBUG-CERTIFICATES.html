// Check store state
const store = window.__VUE_DEVTOOLS_GLOBAL_HOOK__.apps[0]?._instance?.appContext?.config?.globalProperties?.$pinia?._s?.get('certificates')
console.log('üìä Certificates Store State:', {

Common Issues:

  • loading = true forever: Fetch failed but error not caught
  • certificates = []: Database query returned no results
  • error != null: Permission denied or RLS blocking
  • lastFetch = 0: Fetch never completed

Quick Fixes:

1. Force clear loading state:

const store = window.__VUE_DEVTOOLS_GLOBAL_HOOK__.apps[0]?._instance?.appContext?.config?.globalProperties?.$pinia?._s?.get('certificates')
if (store) {
  store.loading = false
  console.log('‚úÖ Force cleared loading state')
}

2. Force refresh data:

const store = window.__VUE_DEVTOOLS_GLOBAL_HOOK__.apps[0]?._instance?.appContext?.config?.globalProperties?.$pinia?._s?.get('certificates')
if (store) {
  store.fetchCertificates(undefined, true)
  console.log('üîÑ Force refreshing certificates')
}

3. Check if you're logged in:

const authStore = window.__VUE_DEVTOOLS_GLOBAL_HOOK__.apps[0]?._instance?.appContext?.config?.globalProperties?.$pinia?._s?.get('auth')
